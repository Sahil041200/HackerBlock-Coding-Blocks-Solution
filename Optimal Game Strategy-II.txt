#include <iostream>
#include <vector>

using namespace std;

int maxWinningValue(vector<int>& coins, int start, int end, vector<vector<int>>& memo) {
    if (start > end) {
        return 0;
    }

    if (memo[start][end] != -1) {
        return memo[start][end];
    }

    int pickStart = coins[start] + min(maxWinningValue(coins, start + 2, end, memo),
                                       maxWinningValue(coins, start + 1, end - 1, memo));

    int pickEnd = coins[end] + min(maxWinningValue(coins, start, end - 2, memo),
                                   maxWinningValue(coins, start + 1, end - 1, memo));

    memo[start][end] = max(pickStart, pickEnd);

    return memo[start][end];
}

int main() {
    int n;
    cin >> n;

    vector<int> coins(n);
    for (int i = 0; i < n; ++i) {
        cin >> coins[i];
    }

    vector<vector<int>> memo(n, vector<int>(n, -1));

    int maxWin = maxWinningValue(coins, 0, n - 1, memo);

    cout << maxWin << endl;

    return 0;
}
