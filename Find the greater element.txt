#include <iostream>
#include <vector>
#include <stack>

using namespace std;

vector<int> findNextGreaterElement(vector<int>& arr) {
    int n = arr.size();
    vector<int> result(n, -1);
    stack<int> s;

    for (int i = 0; i < 2 * n; i++) {
        int current = arr[i % n];
        while (!s.empty() && arr[s.top()] < current) {
            result[s.top()] = current;
            s.pop();
        }
        if (i < n) {
            s.push(i);
        }
    }

    return result;
}

int main() {
    int n;
    cin >> n;

    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    vector<int> result = findNextGreaterElement(arr);

    for (int i = 0; i < n; i++) {
        cout << result[i] << " ";
    }
    cout << endl;

    return 0;
}
