#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

void generateStrings(string& digits, int index, string current, vector<string>& result, const string& mapping) {
    if (index == digits.length()) {
        result.push_back(current);
        return;
    }

    int num = digits[index] - '0';
    if (num >= 1 && num <= 9) {
        generateStrings(digits, index + 1, current + mapping[num - 1], result, mapping);
    }

    if (index + 1 < digits.length()) {
        int twoDigitNum = stoi(digits.substr(index, 2));
        if (twoDigitNum >= 10 && twoDigitNum <= 26) {
            generateStrings(digits, index + 2, current + mapping[twoDigitNum - 1], result, mapping);
        }
    }
}

int main() {
    string digits;
    cin >> digits;
    string mapping = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    vector<string> result;

    generateStrings(digits, 0, "", result, mapping);

    sort(result.begin(), result.end());

    for (string s : result) {
        cout << s << endl;
    }

    return 0;
}
