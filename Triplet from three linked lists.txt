#include<bits/stdc++.h>
using namespace std;

int main() {
    int n, m, k, target;
    cin >> n >> m >> k;
    vector<int> a(n), b(m), c(k);
    for(int i=0; i<n; i++) cin >> a[i];
    for(int i=0; i<m; i++) cin >> b[i];
    for(int i=0; i<k; i++) cin >> c[i];
    cin >> target;
    for(int i=0; i<n; i++)
        for(int j=0; j<m; j++)
            for(int l=0; l<k; l++)
                if(a[i] + b[j] + c[l] == target) {
                    cout << a[i] << " " << b[j] << " " << c[l] << endl;
                    return 0;
                }
    return 0;
}
