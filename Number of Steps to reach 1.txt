#include <iostream>

using namespace std;

int minStepsToOne(int N) {
    if (N == 1) {
        return 0;
    }

    if (N % 2 == 0) {
        return 1 + minStepsToOne(N / 2);
    } else {
        return 1 + min(minStepsToOne(N + 1), minStepsToOne(N - 1));
    }
}

int main() {
    int N;
    cin >> N;

    int result = minStepsToOne(N);

    cout << result << endl;

    return 0;
}
