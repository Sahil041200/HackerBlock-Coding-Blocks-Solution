#include <iostream>
#include <string>
#include <stack>
using namespace std;

string removeDuplicateLetters(string s) {
    int count[26] = {0};
    bool inResult[26] = {false};
    stack<char> st;

    for (char c : s) {
        count[c - 'a']++;
    }

    for (char c : s) {
        count[c - 'a']--;

        if (inResult[c - 'a']) {
            continue;
        }

        while (!st.empty() && c < st.top() && count[st.top() - 'a'] > 0) {
            inResult[st.top() - 'a'] = false;
            st.pop();
        }

        st.push(c);
        inResult[c - 'a'] = true;
    }

    string result = "";
    while (!st.empty()) {
        result = st.top() + result;
        st.pop();
    }

    return result;
}

int main() {
    string s;
    cin >> s;
    string result = removeDuplicateLetters(s);
    cout << result << endl;
    return 0;
}
