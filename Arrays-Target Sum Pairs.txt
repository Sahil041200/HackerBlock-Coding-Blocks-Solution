
#include <bits/stdc++.h>
using namespace std;



int main(){

    int n;
    cin>>n;

    int arr[n];

    for(int i=0; i<n; i++)
    {
        cin>>arr[i];
    }

    int i=0;
    int j=n-1;

    int target;
    cin>>target;

    sort(arr,arr+n);

    while(i<j)
    {
        int sum=arr[i]+arr[j];

        if(sum<target)
        {
            i++;
        }
        else if(sum>target)
        {
            j--;
        }

        else{
            cout<<arr[i]<<" and "<<arr[j]<<endl;
            i++;
            j--;
        }
    }

}
