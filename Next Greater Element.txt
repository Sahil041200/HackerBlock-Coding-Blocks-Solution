#include <bits/stdc++.h>
using namespace std;

// Function to print Next Greater Element for each element of the array
void nextGreater(int arr[], int n)
{
    stack<int> s;
    unordered_map<int, int> nextGreaterMap;

    for (int i = 0; i < n; i++)
    {
        while (!s.empty() && arr[i] > arr[s.top()])
        {
            nextGreaterMap[s.top()] = arr[i];
            s.pop();
        }
        s.push(i);
    }

    while (!s.empty())
    {
        nextGreaterMap[s.top()] = -1;
        s.pop();
    }

    for (int i = 0; i < n; i++)
    {
        cout << arr[i] << ",";
        cout << (nextGreaterMap.count(i) ? nextGreaterMap[i] : -1) << endl;
    }
}

// The Main Function
int main()
{
    int t;
    cin >> t;
    while (t--)
    {
        int n;
        cin >> n;
        int arr[n];
        for (int i = 0; i < n; i++)
        {
            cin >> arr[i];
        }
        nextGreater(arr, n);
    }

    return 0;
}
