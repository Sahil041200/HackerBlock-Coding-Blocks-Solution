#include <iostream>
#include <vector>

using namespace std;

bool isPalindrome(const string& s, int start, int end) {
    while (start < end) {
        if (s[start] != s[end]) {
            return false;
        }
        start++;
        end--;
    }
    return true;
}

int minPartitions(const string& s) {
    int n = s.length();
    vector<int> dp(n, 0);

    for (int i = 1; i < n; ++i) {
        dp[i] = i;

        for (int j = 0; j <= i; ++j) {
            if (isPalindrome(s, j, i)) {
                if (j == 0) {
                    dp[i] = 0;
                } else {
                    dp[i] = min(dp[i], dp[j - 1] + 1);
                }
            }
        }
    }

    return dp[n - 1];
}

int main() {
    int t;
    cin >> t;

    while (t--) {
        string s;
        cin >> s;

        int result = minPartitions(s);

        cout << result << endl;
    }

    return 0;
}
