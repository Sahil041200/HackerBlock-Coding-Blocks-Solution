#include <iostream>
using namespace std;

int lastIndex(int arr[], int N, int M, int index) {
    if (index < 0) {
        return -1; // Base case: M not found in the array
    }

    if (arr[index] == M) {
        return index; // Found M at this index
    }

    // Recursively search in the remaining part of the array
    return lastIndex(arr, N, M, index - 1);
}

int main() {
    int N;
    cin >> N;
    int arr[N];

    for (int i = 0; i < N; i++) {
        cin >> arr[i];
    }

    int M;
    cin >> M;

    int result = lastIndex(arr, N, M, N - 1);
    cout << result << endl;

    return 0;
}
