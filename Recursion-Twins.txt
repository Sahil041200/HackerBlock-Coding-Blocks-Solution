#include <iostream>
using namespace std;

int countTwins(string str, int index) {
    if (index >= str.length() - 2) {
        // If the current index is at the end or beyond, there can't be any twins.
        return 0;
    }

    int twinsCount = 0;

    // Check for twins at the current index
    if (str[index] == str[index + 2] && str[index] != str[index + 1]) {
        twinsCount++;
    }

    // Recursively check for twins starting at the next index
    twinsCount += countTwins(str, index + 1);

    return twinsCount;
}

int main() {
    string input;
    cin >> input;

    int result = countTwins(input, 0);
    cout << result << endl;

    return 0;
}
