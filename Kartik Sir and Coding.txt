#include <iostream>
#include <stack>

using namespace std;

int main() {
    int Q;
    cin >> Q;

    stack<int> modulePile;
    
    for (int i = 0; i < Q; i++) {
        int queryType;
        cin >> queryType;

        if (queryType == 1) {  // Student Query
            if (modulePile.empty()) {
                cout << "No Code" << endl;
            } else {
                int cost = modulePile.top();
                modulePile.pop();
                cout << cost << endl;
            }
        } else if (queryType == 2) {  // Instructor Query
            int cost;
            cin >> cost;
            modulePile.push(cost);
        }
    }

    return 0;
}
