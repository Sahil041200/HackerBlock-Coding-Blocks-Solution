#include <iostream>
using namespace std;

void printLexicographical(int current, int N) {
    if (current > N) {
        return;
    }

    cout << current << " ";

    for (int i = 0; i <= 9; i++) {
        if (10 * current + i <= N && 10 * current + i != 0) {
            printLexicographical(10 * current + i, N);
        }
    }
}

int main() {
    int N;
    cin >> N;

    if (N >= 0) {
        cout << "0 "; 

        for (int i = 1; i <= 9 && i <= N; i++) {
            printLexicographical(i, N);
        }
    }

    cout << endl;

    return 0;
}
