#include <iostream>
#include <climits>

using namespace std;

int main() {
    int num;
    int largest = INT_MIN; // Initialize largest to the smallest possible integer

    for (int i = 0; i < 5; i++) {
        cin >> num;
        if (num > largest) {
            largest = num;
        }
    }

    cout << largest << endl;

    return 0;
}
